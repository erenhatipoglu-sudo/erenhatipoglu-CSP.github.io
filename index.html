4-Bit Binary Converter

This SwiftUI project displays a 4-bit binary representation of numbers from 0 to 15. When the value exceeds 15, it resets back to 0. Each bit is visually represented by colored rectangles — green for 1 and gray for 0.


import SwiftUI

struct ContentView: View {
    @State var id = 0
    let maxID = 15
    
    var body: some View {
        VStack(spacing: 20) {
            Text("Decimal: \(id)")
            HStack(spacing: 10) {
                ForEach(0..<4) { i in
                    let bit = (id >> (3 - i)) & 1
                    Rectangle()
                        .fill(bit == 1 ? Color.green : Color.gray)
                        .frame(width: 40, height: 60)
                }
            }
            HStack(spacing: 20) {
                Button("Enroll (+1)") {
                    id = (id == maxID) ? 0 : (id + 1)
                }
                Button("Reset (0)") {
                    id = 0
                }
            }
        }
        .padding()
    }
}

4-Bit Binary Converter

A Swift project that converts decimal numbers (0–15) into 4-bit binary format.

Show Swift code 

func decimalToBinary(_ number: Int) -> String {
    var remainder = number
    if remainder < 0 {
        remainder = 0
    } else if remainder > 15 {
        remainder = remainder % 16
    }
    let powers = [8, 4, 2, 1]
    var bits = ""
    for power in powers {
        if remainder >= power {
            bits += "1"
            remainder -= power
        } else {
            bits += "0"
        }
    }
    return bits
}

var id = 0
for _ in 1...20 {
    print("Decimal: \(id), Binary: \(decimalToBinary(id))")
    id += 1
    if id > 15 {
        id = 0
    }
}

Explanation: This program displays both decimal and binary values for IDs from 0 to 15. When the number exceeds 15, it resets to 0 (overflow behavior).

SwiftUI Calculator

Simple Calculator i did with Swift code.

Show Swift code 

import SwiftUI
struct ContentView: View {
@State var n1=""
@State var n2=""
@State var result=0
var body: some View {
        VStack {
            Text("Simple Calculator")
            TextField("Enter first number", text:$n1)
            TextField("Enter second number", text:$n2)  
            HStack (spacing:15) {  
                Button("+") {
                    if let a=Int(n1), let b=Int(n2) { result=a+b }
                }
                Button("-") {    
                    if let a=Int(n1), let b=Int(n2) { result=a-b }
                }
                Button("*") {
                    if let a=Int(n1), let b=Int(n2) { result=a*b }
                }
                Button("/") {     
                    if let a=Int(n1), let b=Int(n2) { result=a/b }
                }     
            }
            .buttonStyle(.borderedProminent)
            Text("Result: \(result)")
        }
    }
}

SwiftUI Clock

Clock i did with Swift code.

Show Swift code 

import SwiftUI
struct ContentView: View {
    @State var hour = 13
    @State var minuteSeg = 0   
    var body: some View {
        VStack(spacing:16){
            Text("Time: \(hour)")
            ZStack{
                if hour<12 {
                    Circle().fill(Color.yellow).opacity(0.8).overlay(Circle().stroke(.white,lineWidth:4)).frame(width:200,height:200)
                } else {
                    Circle().fill(Color.blue).opacity(0.8).overlay(Circle().stroke(.white,lineWidth:4)).frame(width:200,height:200)
                }
                Group{
                    if minuteSeg==0 { Rectangle().fill(.white).frame(width:5,height:80).offset(x:0,y:-43) }
                    else if minuteSeg==1 { Rectangle().fill(.white).frame(width:80,height:5).offset(x:43,y:0) }
                    else if minuteSeg==2 { Rectangle().fill(.white).frame(width:5,height:80).offset(x:0,y:43) }
                    else { Rectangle().fill(.white).frame(width:80,height:5).offset(x:-43,y:0) }
                }
                Rectangle().fill(.white).frame(width:8,height:40).offset(x:0,y:-25)
                Circle().fill(.white).frame(width:10,height:10)
            }
            .onTapGesture{
                minuteSeg+=1
                if minuteSeg==4 { minuteSeg=0; hour+=1 }
                if hour==24 { hour=0 }
            }
        }
    }
}

Scratch Tic-Tac-Toe — Documentation

Purpose: Simple two-player Tic-Tac-Toe game built in Scratch for learning game logic and event handling.

How it works:

  • 9 tiles toggle between X and O when clicked.
  • The game checks rows, columns, and diagonals to determine a winner.
  • Highlights the winning line and stops further input once the game ends.

Next steps / Improvements:
