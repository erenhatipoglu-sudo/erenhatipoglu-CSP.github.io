4-Bit Binary Converter

This SwiftUI project displays a 4-bit binary representation of numbers from 0 to 15. When the value exceeds 15, it resets back to 0. Each bit is visually represented by colored rectangles — green for 1 and gray for 0.

Show Swift code 
import SwiftUI

struct ContentView: View {
    @State var id = 0
    let maxID = 15
    
    var body: some View {
        VStack(spacing: 20) {
            Text("Decimal: \(id)")
            HStack(spacing: 10) {
                ForEach(0..<4) { i in
                    let bit = (id >> (3 - i)) & 1
                    Rectangle()
                        .fill(bit == 1 ? Color.green : Color.gray)
                        .frame(width: 40, height: 60)
                }
            }
            HStack(spacing: 20) {
                Button("Enroll (+1)") {
                    id = (id == maxID) ? 0 : (id + 1)
                }
                Button("Reset (0)") {
                    id = 0
                }
            }
        }
        .padding()
    }
}

    }

Scratch Tic-Tac-Toe

A classic Tic-Tac-Toe game made in Scratch. Two players play on the same screen; the game checks rows, columns and diagonals and announces the winner.

How it works:

- The game checks rows, columns and diagonals for a win.

SwiftUI Calculator

Simple Calculator built with SwiftUI: takes two inputs and performs +, −, ×, ÷ operations.

Show Swift code 
import SwiftUI
struct ContentView: View {
    @State var n1 = ""
    @State var n2 = ""
    @State var result = 0

    var body: some View {
        VStack {
            Text("Simple Calculator")
            TextField("Enter first number", text: $n1)
                .textFieldStyle(.roundedBorder)
                .keyboardType(.numberPad)
            TextField("Enter second number", text: $n2)
                .textFieldStyle(.roundedBorder)
                .keyboardType(.numberPad)
            HStack(spacing: 15) {
                Button("+") {
                    if let a = Int(n1), let b = Int(n2) { result = a + b }
                }
                Button("-") {
                    if let a = Int(n1), let b = Int(n2) { result = a - b }
                }
                Button("*") {
                    if let a = Int(n1), let b = Int(n2) { result = a * b }
                }
                Button("/") {
                    if let a = Int(n1), let b = Int(n2), b != 0 { result = a / b }
                }
            }
            .buttonStyle(.borderedProminent)
            Text("Result: \(result)")
        }
        .padding()
    }
}

SwiftUI Clock

Interactive clock built with SwiftUI. Tap the face to advance minute segments; changes appearance for day/night.

Show Swift code 
import SwiftUI

struct ContentView: View {
    @State var hour = 13
    @State var minuteSeg = 0

    var body: some View {
        VStack(spacing: 16) {
            Text("Time: \(hour)")
            ZStack {
                if hour < 12 {
                    Circle()
                        .fill(Color.yellow)
                        .opacity(0.8)
                        .overlay(Circle().stroke(.white, lineWidth: 4))
                        .frame(width: 200, height: 200)
                } else {
                    Circle()
                        .fill(Color.blue)
                        .opacity(0.8)
                        .overlay(Circle().stroke(.white, lineWidth: 4))
                        .frame(width: 200, height: 200)
                }

                Group {
                    if minuteSeg == 0 {
                        Rectangle().fill(.white).frame(width: 5, height: 80).offset(x: 0, y: -43)
                    } else if minuteSeg == 1 {
                        Rectangle().fill(.white).frame(width: 80, height: 5).offset(x: 43, y: 0)
                    } else if minuteSeg == 2 {
                        Rectangle().fill(.white).frame(width: 5, height: 80).offset(x: 0, y: 43)
                    } else {
                        Rectangle().fill(.white).frame(width: 80, height: 5).offset(x: -43, y: 0)
                    }
                }

                Rectangle().fill(.white).frame(width: 8, height: 40).offset(x: 0, y: -25)
                Circle().fill(.white).frame(width: 10, height: 10)
            }
            .onTapGesture {
                minuteSeg += 1
                if minuteSeg == 4 { minuteSeg = 0; hour += 1 }
                if hour == 24 { hour = 0 }
            }
        }
        .padding()
    }
}
